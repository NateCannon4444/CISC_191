#4.2: Input and output
```java
import java.util.Scanner;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.PrintWriter;
import java.io.IOException;

public class InputOutputEditor {
   public static void main(String[] args) throws IOException {
      Scanner scnr = new Scanner(System.in);
      System.out.println("What file is being modified?");
      String inputFileName = scnr.nextLine();
      FileInputStream fileByteStream = new FileInputStream(inputFileName);
      Scanner inFS = new Scanner(fileByteStream);
      FileOutputStream fileStream = new FileOutputStream("ModifiedParks.txt");
      PrintWriter outFS = new PrintWriter(fileStream);
      while (inFS.hasNextLine()) {
         String photoFileName = inFS.nextLine();
         String infoFileName = photoFileName.replace("_photo.jpg", "_info.txt");
         outFS.println(infoFileName);
      }
      outFS.close();
      fileByteStream.close();
      
      System.out.println("New file 'ModifiedParks.txt' has been created.");
   }
}
```

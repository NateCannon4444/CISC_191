import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner keyboard = new Scanner(System.in);
        System.out.println("How many days of temperature data?");
        int days = keyboard.nextInt();
        double[] tempatureArray = new double[days];
        for(int i = 0; i < days; i++){
            System.out.print("Enter the temp for day " + (i + 1) + ": ");
            tempatureArray[i] = keyboard.nextDouble();
        }

        double total = 0;
        for(int i = 0; i < days; i++){
            total += tempatureArray[i];
        }
        double mean = total / days; 
        System.out.println("Mean: " + mean);

            for (int i = 0; i < days - 1; i++) {
                for (int j = 0; j < days - i - 1; j++) {
                    if (tempatureArray[j] > tempatureArray[j + 1]) 
                        {
                        double temp = tempatureArray[j];
                        tempatureArray[j] = tempatureArray[j + 1];
                        tempatureArray[j + 1] = temp;
                    }
                }
            }
            System.out.println("\nSorted Temperatures:");
            for (double t : tempatureArray) {
                System.out.print(t + " ");
            }

        double median = 0.0;
        if (days % 2 != 0) {
            median = tempatureArray[days / 2]; 
        } 
        else {
            median = (tempatureArray[days / 2] + tempatureArray[days / 2 - 1]) / 2.0;
        }

        System.out.println("Median: " + median);

        double max = tempatureArray[0];
        double min = tempatureArray[0];
        for (int i = 1; i <=5; i++)
        {
            if(tempatureArray[i] > max)
            {
                max = tempatureArray[i];
            }
            if(tempatureArray[i] < min)
            {
                min = tempatureArray[i];
            }
        }
        double range = max - min;
        System.out.println("max: "+ max + " min: " + min + " range: " + range);

        double sum = 0;
        for (int i = 0; i < days; i++) {
            double difference = tempatureArray[i] - mean;
            sum += difference * difference;
        }
        double variance = sum / days;
        System.out.println("Variance: " + variance);

        double standardDeviation = Math.sqrt(variance);
        System.out.println("standard deviation: " + standardDeviation);

        if(standardDeviation < 2){
             System.out.println("Temperatures are very consistent");
        }
        else if(standardDeviation < 5){
            System.out.println("Temperatures show moderate variation");
        }
        else{
            System.out.println("Temperatures vary significantly");
        }
    } 
}

int numStudents = keyboard.nextInt();

int[] scores = new int[numStudents];
int scoreSum = 0;
int highest = 0;
int passed = 0;

for (int i = 0; i < numStudents; i++)
{
    System.out.println("Enter score for student " + (i + 1));
    scores[i] = keyboard.nextInt();
    
    scoreSum += scores[i];
    
    if (scores[i] > highest)
    {
        highest = scores[i];
    }
    
    if (scores[i] >= 60)
    {
        passed++;
    }
}

double averageScore = (double) scoreSum / numStudents;

System.out.println("Average score: " + averageScore);
System.out.println("Highest score: " + highest);
System.out.println("Students passed: " + passed);

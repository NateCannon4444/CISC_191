## 6.1: Exceptions
``` Java
import java.util.Scanner;
import java.util.InputMismatchException;

public class Pedometer {
    public static double stepsToMiles(int steps) throws Exception {
        if (steps < 0) {
            throw new Exception("Negative step count entered");
        }
        return steps / 2000.0;
    }
     public static void main(String[] args) {
        Scanner scnr = new Scanner(System.in);
        System.out.println("How many steps?");
        try {

            int userSteps = scnr.nextInt(); 
            double miles = stepsToMiles(userSteps);
            System.out.printf("%.2f\n", miles);         
        } catch (InputMismatchException e) {
            System.out.println("Non-numeric input entered");  
        } catch (Exception e) {
            System.out.println(e.getMessage());
        } finally {
            scnr.close();
        }
    }
}
```
